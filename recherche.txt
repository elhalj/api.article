### Backend : `article.controller.js`
```javascript
import Article from "../models/Article.model.js";

// Récupérer tous les articles publiés (public)
export const getArticles = async (req, res) => {
  try {
    const articles = await Article.find({ status: 'published' })
      .populate('author', 'username')
      .sort({ createdAt: -1 });

    res.status(200).json(articles);
  } catch (error) {
    res.status(500).json({ message: "Erreur serveur", error: error.message });
  }
};

// Récupérer les articles de l'utilisateur connecté
export const getUserArticles = async (req, res) => {
  try {
    const articles = await Article.find({ author: req.user._id })
      .sort({ createdAt: -1 });
      
    res.status(200).json(articles);
  } catch (error) {
    res.status(500).json({ message: "Erreur serveur", error: error.message });
  }
};

// Supprimer un article
export const deleteArticle = async (req, res) => {
  try {
    const article = await Article.findById(req.params.id);

    if (!article) {
      return res.status(404).json({ message: "Article non trouvé" });
    }

    // Vérifier l'auteur ou admin
    if (article.author.toString() !== req.user._id.toString() && req.user.role !== 'admin') {
      return res.status(403).json({ message: "Action non autorisée" });
    }

    await Article.deleteOne({ _id: req.params.id });
    res.status(200).json({ message: "Article supprimé" });
  } catch (error) {
    res.status(500).json({ message: "Erreur serveur", error: error.message });
  }
};

// Modifier un article
export const updateArticle = async (req, res) => {
  try {
    const article = await Article.findById(req.params.id);

    if (!article) {
      return res.status(404).json({ message: "Article non trouvé" });
    }

    if (article.author.toString() !== req.user._id.toString()) {
      return res.status(403).json({ message: "Action non autorisée" });
    }

    const updatedArticle = await Article.findByIdAndUpdate(
      req.params.id,
      { 
        ...req.body,
        status: req.body.status || article.status 
      },
      { new: true, runValidators: true }
    );

    res.status(200).json(updatedArticle);
  } catch (error) {
    res.status(500).json({ message: "Erreur serveur", error: error.message });
  }
};
```

### Frontend : `articleStore.js` (Zustand)
```javascript
import create from 'zustand';
import axios from 'axios';

const useArticleStore = create((set) => ({
  articles: [],
  userArticles: [],
  loading: false,
  error: null,

  // Récupérer les articles publics
  fetchArticles: async () => {
    set({ loading: true, error: null });
    try {
      const response = await axios.get('/api/articles');
      set({ articles: response.data, loading: false });
    } catch (error) {
      set({ error: error.response?.data?.message || error.message, loading: false });
    }
  },

  // Récupérer les articles de l'utilisateur
  fetchUserArticles: async () => {
    set({ loading: true, error: null });
    try {
      const response = await axios.get('/api/articles/me', { withCredentials: true });
      set({ userArticles: response.data, loading: false });
    } catch (error) {
      set({ error: error.response?.data?.message || error.message, loading: false });
    }
  },

  // Créer un article
  createArticle: async (articleData) => {
    set({ loading: true, error: null });
    try {
      const response = await axios.post('/api/articles', articleData, { 
        withCredentials: true 
      });
      set(state => ({
        userArticles: [response.data, ...state.userArticles],
        loading: false
      }));
    } catch (error) {
      set({ error: error.response?.data?.message || error.message, loading: false });
    }
  },

  // Modifier un article
  updateArticle: async (id, updatedData) => {
    set({ loading: true, error: null });
    try {
      const response = await axios.put(`/api/articles/${id}`, updatedData, { 
        withCredentials: true 
      });
      set(state => ({
        userArticles: state.userArticles.map(article =>
          article._id === id ? response.data : article
        ),
        loading: false
      }));
    } catch (error) {
      set({ error: error.response?.data?.message || error.message, loading: false });
    }
  },

  // Supprimer un article
  deleteArticle: async (id) => {
    set({ loading: true, error: null });
    try {
      await axios.delete(`/api/articles/${id}`, { withCredentials: true });
      set(state => ({
        userArticles: state.userArticles.filter(article => article._id !== id),
        loading: false
      }));
    } catch (error) {
      set({ error: error.response?.data?.message || error.message, loading: false });
    }
  }
}));

export default useArticleStore;
```

### Routes backend (à ajouter)
```javascript
// routes/articleRoutes.js
import express from 'express';
import {
  getArticles,
  getUserArticles,
  deleteArticle,
  updateArticle
} from '../controllers/article.controller.js';
import protect from '../middleware/auth.js';

const router = express.Router();

// Public
router.get('/', getArticles);

// Privé
router.get('/me', protect, getUserArticles);
router.delete('/:id', protect, deleteArticle);
router.put('/:id', protect, updateArticle);

export default router;
```

#### 3. Middleware d'authentification :
```javascript
export const protect = async (req, res, next) => {
  const token = req.cookies.jwt;
  if (!token) return res.status(401).json({ message: 'Non authentifié' });

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = await User.findById(decoded.userId);
    next();
  } catch (error) {
    res.status(401).json({ message: 'Token invalide' });
  }
};
```

```javascript
import create from 'zustand';

const useAuthStore = create((set) => ({
  user: null,
  login: async (credentials) => {
    const res = await axios.post('/api/auth/login', credentials);
    set({ user: res.data });
  },
  logout: () => {
    axios.post('/api/auth/logout');
    set({ user: null });
  }
}));
```

#### 3. ArticleForm.jsx :
```javascript
function ArticleForm() {
  const [formData, setFormData] = useState({ title: '', content: '' });
  const { createArticle } = useArticleStore();

  const handleSubmit = async (e) => {
    e.preventDefault();
    await createArticle(formData);
    setFormData({ title: '', content: '' });
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="text"
        placeholder="Titre"
        value={formData.title}
        onChange={(e) => setFormData({ ...formData, title: e.target.value })}
      />
      <textarea
        placeholder="Contenu"
        value={formData.content}
        onChange={(e) => setFormData({ ...formData, content: e.target.value })}
      />
      <button type="submit">Publier</button>
    </form>
  );
}
```

### **Flux d'authentification**
1. L'utilisateur s'inscrit/login
2. Le backend envoie un cookie HTTP-only avec le JWT
3. Zustand met à jour le state d'authentification
4. Le dashboard charge les articles de l'utilisateur
5. La création d'article envoie le JWT automatiquement via les cookies
